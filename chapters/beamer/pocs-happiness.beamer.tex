    Build on measures of the emotional content of \alertb{individual words}.
  \item<2->
    Osgood et al. (1957)\cite{osgood1957a} identified\\ a basis of three

    \begin{itemize}
    \item<3->
      \alert{Valence:} \alertb{bad $\leftrightarrow$ good }
    \item<3->
      \alert{Dominance:} \alertb{weak $\leftrightarrow$ strong }
    \item<3->
      \alert{Arousal:} \alertb{passive $\leftrightarrow$ active }
    \end{itemize}

\begin{frame}
  \frametitle{Valence shift details:}

  \begin{block}{Given two texts $a$ and $b$:}
    \begin{itemize}
    \item<1->
      Measure difference in average valence: $v_{avg}^{(b)} - v_{avg}^{(a)}$
    \item<2->  
      Break difference down by contributions from individual words:
      $$
      \Delta_i
      =100 \times
      [p_{i,b} - p_{i,a}]
      \frac{
      [v_i - v_{\textrm{avg}^{(a)}]
      }
      {
      [v_{\textrm{avg}^{(b)}-v_{\textrm{avg}^{(a)}]
    }
      $$
      $$
      \sum_{i} \Delta_i = v_{avg}^{(b)} - v_{avg}^{(a)}
      $$
    \item<2->  
      Rank words by $|\Delta_i|$
    \end{itemize}
  \end{block}

\end{frame}
