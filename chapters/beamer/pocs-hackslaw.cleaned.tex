  \textsl{\small Fluctuations in Hack's law:}\hfill\mbox{}\\
  \vspace{5mm}
  Mean Hack's law:  $ \avg{l} = \theta a^h $

  Scaling ansatzes:
  $$\cprob{l}{a} = 1/a^h F_l (l/a^h)$$
  $$\cprob{a}{l} = 1/l^{1/h} F_a (a/l^{1/h})$$
  $$P(a,l) = P(a) \cprob{l}{a} = P(l)\cprob{a}{l}$$

  $$P(a) \sim N_a a^{-\tau} 
  \quad
  \mbox{and}
  \quad
  P(l) \sim N_l l^{-\gamma}$$

  $$\tau = 2 - h
  \quad
  \mbox{and}
  \quad
  \gamma = 1/h$$

   \textbf{Fluctuations in Hack's law:}

  Scaling form of $P(a,l)$
  $$P(a,l) = 1/a^{2} F(l/a^h) = 1/l^{2/h} G(a/l^{1/h})$$
where\\
$x^{-2} F(x^{-h}) = G(x)$ and
$F(y) = y^{-2} G(y^{-1/h})$

%From random walks:
%$$P(n) \sim \frac{1}{\sqrt{2}\pi} n^{-3/2}
%\rightarrow P(l) \sim \frac{2}{\pi} l^{-3/2}$$

Postulate:
$$  \cprob{l}{a} = \frac{1}{\sqrt{2\pi} a^{2/3} \eta}
  \exp\{-(l/a^{2/3}-\theta)^2/2\eta^2\} $$

$$
  P(a,l)  =  \frac{N_a}{a^{4/3}}
  \frac{1}{\sqrt{2\pi} a^{2/3} \eta}
  \exp\{-(l/a^{2/3}-\theta)^2/2\eta^2\} $$
$$ = 
  \frac{N_a}{\sqrt{2\pi} a^{2} \eta}
  \exp\{-(l/a^{2/3}-\theta)^2/2\eta^2\} $$
$$     N_a =  \frac{2^{5/2}\eta}{\sqrt{3\pi}}
  \left[ \int_{u=0}^\infty \dee{u}
  u^{1/2} \exp\{-(u-\theta)^2/2\eta^2\} \right]^{-1}$$

   

  \textsl{\small Fluctuations in Hack's law:}\hfill\mbox{}\\
  \vspace{5mm}
  $$ F_l(z)  =  \frac{1}{\sqrt{2\pi} \eta}
  \exp\{-(z-\theta)^2/2\eta^2\} $$

  $$ F_a(z)  =  \frac{N_a \sqrt{\pi}}{2^{3/2}\eta} z^{-2}
  \exp\{-(z^{-2/3}-\theta)^2/2\eta^2\} $$

  $$ F(z)  =  \frac{N_a}{\sqrt{2\pi} \eta}
  \exp\{-(z-\theta)^2/2\eta^2\} $$

  $$ G(z)  =  \frac{2N_a}{\sqrt{\pi}2^{3/2}\eta} z^{-2}
  \exp\{-(z^{-2/3}-\theta)^2/2\eta^2\} $$

   

  \textsl{\small Fluctuations in Hack's law:}\hfill\mbox{}\\
  \vspace{5mm}
  $ \cprob{l}{a} = 1/a^h F_l(l/a^h)$\\
  \mbox{}\\
  $  = 1/a^h \frac{1}{\sqrt{2\pi} \eta} \exp\{-(l/a^h-\theta)^2/2\eta^2\}$

  $\cprob{a}{l} = 1/l^{1/h} F_a(a/l^{1/h})$\\
  \mbox{}\\
  $=  1/l^{1/h} \frac{N_a}{\sqrt{2\pi} N_l \eta} 
  (a/l^{1/h})^{-2}$\\
  $\qquad \times \exp\{-((a/l^{1/h})^{-h}-\theta)^2/2\eta^2\},$

the full Hack distribution:
$ P(a,l)  =  1/a^{2} F(l/a^h) = 1/l^{2/h} G(a/l^{1/h})$\\
  \mbox{}\\
 $ = \frac{N_a}{\sqrt{2\pi} \eta a^2} \exp\{-(l/a^h-\theta)^2/2\eta^2\}.$
